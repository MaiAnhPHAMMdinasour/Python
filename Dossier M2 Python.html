def depnai(Deph, Depf):
    if (Deph == '59') and (Depf == '59'):
        return 'Couple du Nord'
    elif (Deph == '59') and (Depf != '59'):
        return 'Homme du Nord - Femme de loin'
    elif (Deph != '59') and (Depf == '59'):
        return 'Femme du Nord - Homme de loin'
    else:
        return 'Couple de loin'
